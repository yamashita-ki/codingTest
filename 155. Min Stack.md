class MinStack {
    
    private final Deque<Integer> stack;

    private int min;

    public MinStack() {
        stack = new ArrayDeque<>();
        min = Integer.MAX_VALUE;
    }
    
    public void push(int val) {
        stack.add(val);
        min = Math.min(min, val);
    }
    
    public void pop() {
        stack.pollLast();
        // ここでminをどうするかが実装できず
    }
    
    public int top() {
        return stack.getLast();
    }
    
    public int getMin() {
        return min;
    }
}
